create table lecture_info(
no number primary key,
tag varchar2(30) not null,
title varchar2(100) not null,
teacher varchar2(20) not null,
picture_name varchar2(256),
picture_realname varchar2(256),
picture_type varchar2(100),
picture_size number,
intro varchar2(300),
detail varchar2(4000) not null,
possible varchar2(3) not null,
price number not null,
accept varchar2(3) not null,
reg date not null,

);

create sequence lecture_info_seq nocache;
